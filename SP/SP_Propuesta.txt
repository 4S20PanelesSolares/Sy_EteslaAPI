CREATE PROCEDURE SP_Propuesta
(
	IN Opcion INT,
	/*Variables de Cliente*/
	IN xIdCliente VARCHAR(255),
	/*Variables para Propuesta*/
	IN xIdPropuesta VARCHAR(255),
	IN xIdPanel VARCHAR(255),
	IN xIdInversor VARCHAR(255),
	IN xIdUsuario VARCHAR(255),
	IN TipoCotizacion CHAR(12),
	IN Tarifa CHAR(10),
	IN CantidadPaneles INT,
	IN CantidadInversores INT,
	IN PotenciaPropuesta FLOAT,
	IN GeneracionPaneles FLOAT,
	IN NuevoConsumoCFE FLOAT,
	IN Descuento INT,
	IN Subtotal FLOAT,
	IN PrecioConIVA FLOAT,
	IN Total FLOAT,
	IN StatusProjectFV TINYINT,
	IN daysOfExpire INT
)
BEGIN
	DECLARE expireAt TIMESTAMP;

	IF(Opcion = 0) THEN
		/*Insertar*/
		UPDATE cliente
		SET bTienePropuesta = 1
		WHERE idCliente = UNHEX(xIdCliente);
		/*-----------------------------*/
		SET expireAt = DATE_ADD(NOW(),INTERVAL daysOfExpire DAY);
		INSERT INTO Propuesta
		VALUES(UUID(), UNHEX(xIdPanel), UNHEX(xIdInversor), UNHEX(xIdCliente), UNHEX(xIdUsuario), TipoCotizacion, Tarifa, CantidadPaneles, CantidadInversores, PotenciaPropuesta, GeneracionPaneles, NuevoConsumoCFE, Descuento, Subtotal, PrecioConIVA, Total, StatusProjectFV, now(), null, null, expireAt);
	ELSEIF (Opcion = 1) THEN
		/*Borrado logico*/
		UPDATE cliente
		SET deleted_at = now()
		WHERE idPropuesta = UNHEX(xIdPropuesta);
	ELSEIF (Opcion = 2) THEN
		/*Editar / Actualizar*/
		UPDATE Propuesta
		SET id_Panel= HEX(xIdPanel), id_Inversor = HEX(xIdInversor), cTipoCotizacion = TipoCotizacion, cTarifa = Tarifa, iCantidadPaneles = CantidadPaneles, iCantidadInversores = CantidadInversores, fPotencialPropuesta = PotenciaPropuesta, fGeneracionPaneles = GeneracionPaneles, fNuevoConsumoCFE = NuevoConsumoCFE, iPorcentajeDescuento = Descuento, fSubtotal = Subtotal, fprecioConIVA = PrecioConIVA, fTotal = Total, updated_at = now();
	ELSEIF (Opcion = 3) THEN
		/*Leer*/
		SELECT HEX(idPropuesta), HEX(id_Panel), HEX(id_Inversor), HEX(id_Cliente), HEX(id_Usuario), cTipoCotizacion, cTarifa, iCantidadPaneles, iCantidadInversores, fPotencialPropuesta, fGeneracionPaneles, fNuevoConsumoCFE, Descuento, fSubtotal, fprecioConIVA, fTotal, siStatusProjejctFV, created_at, deleted_at, updated_at, expire_at
		FROM Propuesta
		WHERE deleted_at IS NULL;
	ELSEIF (Opcion = 4) THEN
		/*Consulta_filtrada*/
		SELECT HEX(idPropuesta), HEX(id_Panel), HEX(id_Inversor), HEX(id_Cliente), HEX(id_Usuario), cTipoCotizacion, cTarifa, iCantidadPaneles, iCantidadInversores, fPotencialPropuesta, fGeneracionPaneles, fNuevoConsumoCFE, Descuento, fSubtotal, fprecioConIVA, fTotal, siStatusProjejctFV, created_at, deleted_at, updated_at, expire_at
		FROM Propuesta
		WHERE idPropuesta = HEX(xIdPropuesta);
	END IF;
END
// DELIMITER;
